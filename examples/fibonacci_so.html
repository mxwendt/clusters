/**
         * @cluster
         *
         * @param {Number} n = 5, 2, 10
         */
        function fibo(n) {
            var resultsArray = [];

            for (var i = 0; i <= n; i++) {
                if (i === 0) {
                    resultsArray.push(0);
                } else if (i === 1) {
                    resultsArray.push(1);
                } else {
                    resultsArray.push(resultsArray[i - 2] + resultsArray[i - 1]);
                }
            }

            return resultsArray[n];
        }
