/**
     * @param {Object} legend['#'] = {}
     * @param {Object} legend['o'] = {}
     * @param {String} ch = "#", "o", " "
     */
    function elementFromChar(legend, ch) {
        if (ch == ' ') {
            return null;
        }
        var element = new legend[ch]();
        element.originChar = ch;
        return element;
    }
